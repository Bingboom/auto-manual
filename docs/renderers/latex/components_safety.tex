\providecommand{\safetywarning}[1]{}
\renewcommand{\safetywarning}[1]{%
  \begin{safetywarningbox}%
    \noindent
    \begin{tabular}{@{}m{\csname HBcomp_lockup_width\endcsname}%
                    @{\hspace{\csname HBcomp_lockup_gap\endcsname}}%
                    m{\dimexpr\textwidth-\csname HBcomp_lockup_width\endcsname-\csname HBcomp_lockup_gap\endcsname\relax}@{}}%
      \centering\warninglockup
      &
      \parbox[c]{\dimexpr\textwidth-\csname HBcomp_lockup_width\endcsname-\csname HBcomp_lockup_gap\endcsname\relax}{%
        \HBTypeWarningTextStart
        \HBapplyalignment
        \HBtightlines{\csname HBtype_warning_text_font_leading\endcsname}{%
          \HBifupper{\csname HBtype_warning_text_force_upper\endcsname}{#1}%
        }%
      }%
    \end{tabular}%
  \end{safetywarningbox}%
}

  \expandafter\def\csname HB#1\endcsname{#2}%
}

\providecommand{\HBOverrideFromKey}[2]{%
  \ifcsname HB#2\endcsname
    \expandafter\ifx\csname HB#2\endcsname\relax
      % Skip undefined placeholder keys.
    \else
      \expandafter\edef\csname HB#1\endcsname{\csname HB#2\endcsname}%
    \fi
  \fi
}
