\setlength{\columnsep}{\csname HBcomp_twocol_sep\endcsname}

% Spacing is declared, not cancelled: \multicolsep (multicol's own 12pt
% glue before/after the columns — the actual source the old negative
% comp_twocol_tighten was fighting, not "Sphinx glue") is zeroed inside
% the environment group, and comp_twocol_before / comp_twocol_after state
% the intended gaps directly (typography_gap/2026-07-03 #11).
\newenvironment{safetytwocol}{%
  \par\noindent
  \setlength{\multicolsep}{0pt}%
  \vspace*{\csname HBcomp_twocol_before\endcsname}%
  \begin{multicols}{2}
    \HBTypeListStart
    \RaggedRight
    \hyphenpenalty=10000\relax
    \exhyphenpenalty=10000\relax
}{%
  \end{multicols}
  \par\vspace*{\csname HBcomp_twocol_after\endcsname}
}
