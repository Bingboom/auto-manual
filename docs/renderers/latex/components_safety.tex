\providecommand{\safetywarning}[1]{}
\renewcommand{\safetywarning}[1]{%
  \begin{safetywarningbox}%
    \noindent
    \begin{tabular}{@{}m{\csname HBcomp_lockup_width\endcsname}%
                    @{\hspace{\csname HBcomp_lockup_gap\endcsname}}%
                    m{\dimexpr\textwidth-\csname HBcomp_lockup_width\endcsname-\csname HBcomp_lockup_gap\endcsname\relax}@{}}%
      \centering\warninglockup
      &
      \parbox[c]{\dimexpr\textwidth-\csname HBcomp_lockup_width\endcsname-\csname HBcomp_lockup_gap\endcsname\relax}{%
        \HBTypeWarningTextStart
        \HBapplyalignment
        \HBtightlines{\csname HBtype_warning_text_font_leading\endcsname}{%
          \HBifupper{\csname HBtype_warning_text_force_upper\endcsname}{#1}%
        }%
      }%
    \end{tabular}%
  \end{safetywarningbox}%
}

\providecommand{\HBWarningLeadTriangle}{%
  \begin{tikzpicture}[x=1mm,y=1mm]
    \draw[line width=0.55pt, line join=round, color=BrandDark]

    \node[BrandDark,font=\fontsize{8.5pt}{8.5pt}\selectfont\bfseries] at (4.6,2.6) {!};

\providecommand{\HBWarningLeadBlock}[2]{%
  % Column-safe: sized off \linewidth (not \textwidth) so the block can
  % sit inside safetytwocol's left column, matching the V2.0 master p4.
  \par\noindent
  \begin{tcolorbox}[
    enhanced,
    colback=white,
    colframe=LineK40,
    boxsep=\csname HBcomp_warning_box_sep\endcsname,
    arc=\csname HBcomp_warning_box_arc\endcsname,
    boxrule=\csname HBcomp_warning_box_rule\endcsname,
    left=\csname HBcomp_warning_box_pad_lr\endcsname,
    right=\csname HBcomp_warning_box_pad_lr\endcsname,
    top=0.8mm, bottom=0.8mm,
    before skip=\csname HBcomp_warning_box_before\endcsname,
    after skip=\csname HBcomp_warning_box_after\endcsname,
  ]
    \noindent
    \begin{tabular}{@{}m{9.5mm}@{\hspace{1.6mm}}m{\dimexpr\linewidth-9.5mm-1.6mm-2\tabcolsep\relax}@{}}%
      \centering\HBWarningLeadTriangle
      &
      {\HBFontHeavy\color{\csname HBtype_body_color\endcsname}%
       \fontsize{8pt}{9pt}\selectfont #1}%
      \par\vspace{0.4mm}%
      {\HBTypeWarningTextStart #2}%
      \tabularnewline
    \end{tabular}%
  \end{tcolorbox}%
  \par
}

      &
      \parbox[c]{\dimexpr\textwidth-\csname HBcomp_lockup_width\endcsname-\csname HBcomp_lockup_gap\endcsname\relax}{%
        \HBTypeWarningTextStart
        \HBapplyalignment
        \HBtightlines{\csname HBtype_warning_text_font_leading\endcsname}{%
          #2%
          \HBIfBlankTF{#3}{}{\par #3}%
        }%
      }%
    \end{tabular}%
  \end{safetywarningbox}%
}

\setlength{\columnsep}{\csname HBcomp_twocol_sep\endcsname}

% Spacing is declared, not cancelled: \multicolsep (multicol's own 12pt
% glue before/after the columns — the actual source the old negative
% comp_twocol_tighten was fighting, not "Sphinx glue") is zeroed inside
% the environment group, and comp_twocol_before / comp_twocol_after state
% the intended gaps directly (typography_gap/2026-07-03 #11).
\newenvironment{safetytwocol}{%
  \par\noindent
  \setlength{\multicolsep}{0pt}%
  \vspace*{\csname HBcomp_twocol_before\endcsname}%
  \begin{multicols}{2}
    \HBTypeListStart
}{%
  \end{multicols}
  \par\vspace*{\csname HBcomp_twocol_after\endcsname}
}
