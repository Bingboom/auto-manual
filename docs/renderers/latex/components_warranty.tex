\providecommand{\HBWarrantyYearBadge}[1]{%
  \tikz[baseline=(HBWarrantyNumber.base)]{%
    \node[
      circle,
      fill=BrandDark,
      text=white,
      inner sep=0pt,
      minimum size=\csname HBcomp_warranty_year_badge_size\endcsname,
      font=\HBFontHeavy
        \fontsize{\csname HBtype_warranty_year_number_font_size\endcsname}
                 {\csname HBtype_warranty_year_number_font_size\endcsname}\selectfont,
    ] (HBWarrantyNumber) {#1};%
  }%
}

  \noindent

  \hspace{1.4mm}%
  \begin{minipage}[c]{\dimexpr\linewidth
    -\csname HBcomp_warranty_year_badge_size\endcsname-1.4mm\relax}
    {\HBFontHeavy
     \fontsize{\csname HBtype_warranty_year_unit_font_size\endcsname}
              {\csname HBtype_warranty_year_unit_font_size\endcsname}\selectfont
     #2\par}%
    \vspace{0.15mm}%

     \fontsize{\csname HBtype_warranty_year_subtitle_font_size\endcsname}
              {\csname HBtype_warranty_year_subtitle_font_size\endcsname}\selectfont
     #3\par}%
  \end{minipage}%
  \par\vspace{0.8mm}%
}
