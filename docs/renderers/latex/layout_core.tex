% ==================================================
% L4 CORE LAYOUT MECHANISM
% ==================================================

\usepackage{geometry}
\usepackage{fancyhdr}

% ==========================
% PAGE GEOMETRY
% ==========================

% --- fallback to avoid 0bp paper size on some MiKTeX setups ---
\expandafter\providecommand\csname HBpage_paperwidth\endcsname{130.10mm}
\expandafter\providecommand\csname HBpage_paperheight\endcsname{185.10mm}


\geometry{
  paperwidth=\csname HBpage_paperwidth\endcsname,
  paperheight=\csname HBpage_paperheight\endcsname,
  left=\csname HBpage_margin_left\endcsname,
  right=\csname HBpage_margin_right\endcsname,
  top=\csname HBpage_margin_top\endcsname,
  bottom=\csname HBpage_margin_bottom\endcsname
}

% ==========================
% BASE PAGE STYLE (EMPTY)
% ==========================

\fancypagestyle{HBEmptyStyle}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

% ==========================
% GLOBAL PARAGRAPH
% ==========================

\setlength{\parindent}{\csname HBpage_parindent\endcsname}
\setlength{\parskip}{\csname HBpage_parskip\endcsname}

% ==========================
% GLOBAL LINE BREAKING
% TeX defaults hyphenate aggressively in the narrow 130mm measure
% (typography_gap/2026-07-03 #9/#10). Raise the hyphenation cost and
% give justification an emergency pass so word spacing stays even.
% Values are tuned in data/layout_params.csv (LINE BREAKING block).
% ==========================
\expandafter\providecommand\csname HBtext_hyphenpenalty\endcsname{800}
\expandafter\providecommand\csname HBtext_doublehyphendemerits\endcsname{10000}
\expandafter\providecommand\csname HBtext_finalhyphendemerits\endcsname{8000}
\expandafter\providecommand\csname HBtext_tolerance\endcsname{1500}
\expandafter\providecommand\csname HBtext_emergencystretch\endcsname{1.0em}
\hyphenpenalty=\csname HBtext_hyphenpenalty\endcsname\relax
\doublehyphendemerits=\csname HBtext_doublehyphendemerits\endcsname\relax
\finalhyphendemerits=\csname HBtext_finalhyphendemerits\endcsname\relax
\tolerance=\csname HBtext_tolerance\endcsname\relax
\setlength{\emergencystretch}{\csname HBtext_emergencystretch\endcsname}

% ==================================================
% PRINT FOOTER: odd-left / even-right, 01 02 ...
% This MUST live in theme (NOT conf), and must override plain.
% ==================================================

% two-digit page number helper (01..)
\newcommand{\HBpagenum}{%
  \fontsize{\csname HBpage_footer_font_size\endcsname}
           {\csname HBpage_footer_font_leading\endcsname}\selectfont
  \ifnum\csname HBpage_footer_leading_zero\endcsname=1\relax
    \ifnum\value{page}<10 0\fi
  \fi
  \arabic{page}%
}

% Normal style (main body pages)
\fancypagestyle{normal}{%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%

  % clear all positions to avoid any residual center page number
  \fancyfoot[C]{}%
  \fancyfoot[CO]{}%
  \fancyfoot[CE]{}%
  \fancyfoot[LO]{}%
  \fancyfoot[LE]{}%
  \fancyfoot[RO]{}%
  \fancyfoot[RE]{}%

  % odd pages -> left bottom
  \fancyfoot[LO]{\raisebox{\csname HBpage_footer_raise\endcsname}{\HBpagenum}}%

  % even pages -> right bottom
  \fancyfoot[RE]{\raisebox{\csname HBpage_footer_raise\endcsname}{\HBpagenum}}%
}

% Plain style (Sphinx/LaTeX often uses this on section/chapter first pages)
\fancypagestyle{plain}{%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%

  % clear all positions
  \fancyfoot[C]{}%
  \fancyfoot[CO]{}%
  \fancyfoot[CE]{}%
  \fancyfoot[LO]{}%
  \fancyfoot[LE]{}%
  \fancyfoot[RO]{}%
  \fancyfoot[RE]{}%

  \fancyfoot[LO]{\raisebox{\csname HBpage_footer_raise\endcsname}{\HBpagenum}}%
  \fancyfoot[RE]{\raisebox{\csname HBpage_footer_raise\endcsname}{\HBpagenum}}%
}

% KILL default centered plain forever (critical!)
\makeatletter
\let\ps@plain\ps@normal
\makeatother
