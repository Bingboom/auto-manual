\providecommand{\HBBackCoverPage}[3]{%
  \clearpage

}

% V2.0 master: each item is its own rounded card with a circled index
% at the top-left, the line art centered, and the label underneath.
\providecommand{\HBInBoxCard}[3]{%
  \begin{tcolorbox}[
    enhanced,
    colback=white,
    colframe=LineK40,
    arc=\csname HBcomp_inbox_card_arc\endcsname,
    boxrule=\csname HBcomp_table_outer_rule\endcsname,
    boxsep=0mm,
    left=1.2mm,right=1.2mm,top=5.8mm,bottom=2.0mm,
    width=\csname HBcomp_inbox_card_width\endcsname\linewidth,
    height=\csname HBcomp_inbox_card_height\endcsname,
    nobeforeafter,
    equal height group=hbinbox,
    valign=center,
    overlay={
      \node[fill=BrandDark,circle,inner sep=0.65mm,
            font=\HBFontHeavy\fontsize{6.5pt}{6.5pt}\selectfont,
            text=white]
        at ([yshift=-4.2mm]frame.north) {#1};
    },
  ]
    {\centering
     \parbox[c][\csname HBcomp_inbox_image_height\endcsname][c]
       {\linewidth}{\centering
         \HBImageOrPlaceholder
           {\csname HBcomp_inbox_image_width\endcsname}
           {\csname HBcomp_inbox_image_height\endcsname}
           {#2}\par}\par}%
    \vspace{\csname HBcomp_inbox_label_gap\endcsname}%
    {\centering\HBTypeInBoxLabel{#3}\par}%
  \end{tcolorbox}%
}

\newcommand{\HBOverviewPanel}[3]{%
  \HBTitleLevelTwoBlock{#1}%

    \HBImageOrPlaceholder
      {\csname HBcomp_overview_image_width\endcsname}
      {\csname HBcomp_overview_image_height\endcsname}
      {#2}%
  \end{center}
  \vspace{-\csname HBcomp_overview_image_after\endcsname}%
  \par\noindent
  \begin{hboverviewtableframe}
  \begingroup
  \arrayrulecolor{HBSharedTableFrameColor}
  \arrayrulewidth=\dimexpr\csname HBcomp_table_outer_rule\endcsname/2\relax
  \ifdim\arrayrulewidth<0.2pt\arrayrulewidth=0.2pt\fi

    >{\RaggedRight\arraybackslash}m{\dimexpr(\linewidth-4\tabcolsep-3\arrayrulewidth)/2\relax}|
    >{\RaggedRight\arraybackslash}m{\dimexpr(\linewidth-4\tabcolsep-3\arrayrulewidth)/2\relax}|}
  \hline
  #3%
  \end{tabular}
  \endgroup
  \end{hboverviewtableframe}
  \par\vspace{0.8mm}%
